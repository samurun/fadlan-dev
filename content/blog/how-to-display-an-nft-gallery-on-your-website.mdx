เมื่อไม่นานมานี้ได้มีโอกาสทำโปรเจกต์เกี่ยวกับ Smart contract โดยการเอา NFTs มาแสดงบนเว็ป เลยอยากเอามาแชร์เผื่อมีคนอยากรู้ โดยเครื่องมือจะมีดังนี้

- [Vite](https://vitejs.dev/) ใช้สำหรับสร้างโปรเจกต์และจัดการโครงสร้างไฟล์
- [Tailwindcss](https://tailwindcss.com/) ใช้สำหรับการจัดการสไตล์ของหน้าเว็บ
- [TanStack Query](https://tanstack.com/query/latest) สำหรับจัดการการดึงข้อมูลแบบ asynchronous
- [Viem](https://viem.sh) ใช้ในการเชื่อมต่อและทำงานกับ blockchain

ผมจะข้ามขั้นตอนการ setup นะครับ ส่วนโครงสร้างโฟลเดอร์มีดังนี้:


## 1 การแสดงผล NFT tokens

โดย NFT ที่จะนำมาแสดงเอาจาก Bitkub ตามลิงค์นี้:

```url
https://next-api.bitkubnft.com/api/nft-tokens
```

การสร้าง Token List Component

```tsx
//src/components/token-list.tsx

import { ticketAbi } from '@/lib/abis/ticketAbi';
import { BKMainnet } from '@/lib/BKMainnet';
import { WALLET_ADDRESS } from '@/lib/walletAddress';
import { NFTToken } from '@/types';
import { useQuery } from '@tanstack/react-query';
import { useCallback, useEffect, useState } from 'react';
import { Address } from 'viem';
import { Checkbox } from './ui/checkbox';
import TokenItem from './token-item';
import { Button } from './ui/button';
import SkeletonWrapper from './skeleton-wrapper';
import { useLocation, useNavigate } from 'react-router';
import { useSearchParams } from 'react-router-dom';

type Props = {};

// Function to fetch balance
export const fetchBalance = async (address: string, walletAddress: string) => {
  return BKMainnet.readContract({
    address: address as Address,
    abi: ticketAbi,
    functionName: 'balanceOf',
    args: [walletAddress as Address],
  });
};

const NFTList = ({}: Props) => {
  const [searchParams] = useSearchParams();
  const navigate = useNavigate();
  const location = useLocation();
  const [balances, setBalances] = useState<Record<string, string>>({});
  const [loading, setLoading] = useState<boolean>(true);

  const { data: tokens, isLoading: tokenLoading } = useQuery({
    queryKey: ['nft-tokens'],
    queryFn: () =>
      fetch('https://next-api.bitkubnft.com/api/nft-tokens')
        .then((res) => res.json())
        .then((res) => res.data as Array<NFTToken>),
  });

  useEffect(() => {
    const fetchAllBalances = async () => {
      setLoading(true);
      const mappedTokens = tokens?.map((token) => {
        if (token.nfts) {
          return token.nfts.map((nft) => nft.address as String);
        }

        return token.address as String;
      });

      try {
        const balancePromises: Promise<void>[] = [];

        mappedTokens?.forEach((token) => {
          if (Array.isArray(token)) {
            token.forEach((nestedToken) => {
              balancePromises.push(
                fetchBalance(String(nestedToken), WALLET_ADDRESS).then(
                  (balance) => {
                    setBalances((prevBalances) => ({
                      ...prevBalances,
                      [nestedToken as any]: balance,
                    }));
                  }
                )
              );
            });
          } else {
            balancePromises.push(
              fetchBalance(String(token), WALLET_ADDRESS).then((balance) => {
                setBalances((prevBalances) => ({
                  ...prevBalances,
                  [token as any]: balance,
                }));
              })
            );
          }
        });

        await Promise.all(balancePromises);
      } catch (err) {
        console.error('Error fetching balances:', err);
      } finally {
        setLoading(false);
      }
    };

    if (tokens) {
      fetchAllBalances();
    }
  }, [tokens]);

  const filteredMyNFTOnly = useCallback(() => {
    if (searchParams.has('show-nft')) {
      return tokens?.filter((token) => {
        const balanceOf =
          token.type === 'group'
            ? token.nfts
                ?.map((nft) => Number(`${balances[nft.address]}` || 0))
                .reduce((acc, curr) => acc + curr, 0) || 0
            : Number(`${balances[token.address]}` || 0);

        return balanceOf > 0;
      });
    }

    return tokens;
  }, [tokens, balances, searchParams]);

  return (
    <div>
      <section className='flex justify-between'>
        <h1 className=' font-medium'>NFTs token</h1>
        <Button>Add Walletaddress</Button>
      </section>
      <div className='flex items-center space-x-2 mt-4'>
        <Checkbox
          checked={searchParams.has('show-nft')}
          id='terms'
          onCheckedChange={() => {
            const params = new URLSearchParams(searchParams);
            searchParams.has('show-nft')
              ? params.delete('show-nft')
              : params.set('show-nft', 'true');
            navigate(`${location.pathname}?${params}`);
          }}
        />
        <label
          htmlFor='terms'
          className='text-sm font-medium leading-none peer-disabled:cursor-not-allowed peer-disabled:opacity-70'
        >
          Show My NFTs Only
        </label>
      </div>
      <div className='space-y-2 mt-4'>
        {filteredMyNFTOnly()?.map((project) => (
          <TokenItem
            key={project.address}
            address={project.address}
            name={project.name}
            image={project.image}
            type={project.type}
            nfts={project.nfts}
            balances={balances}
          />
        ))}
      </div>
    </div>
  );
};

export default NFTList;
```

การสร้าง Token Item Component

```jsx
import { Nft } from '@/types';

type Props = {
  address: string;
  name: string;
  image: string;
  type: string;
  nfts?: Array<Nft>;
  balances: Record<string, string>;
};

const TokenItem = ({ address, name, image, type, nfts, balances }: Props) => {
  return (
    <div
      key={address + name}
      className='border p-2 rounded flex flex-col gap-2 relative'
    >
      <div className='flex gap-2'>
        <img
          src={image}
          className='size-8 object-contain rounded overflow-hidden'
        />
        {type === 'group' ? (
          <p>{name}</p>
        ) : (
          <div className='flex items-center'>
            <a href={`${address}`}>
              {name}{' '}
              <span className=' absolute right-4'>{`${
                balances![address] || 0
              }`}</span>
            </a>
          </div>
        )}
        {nfts?.length ? (
          <p className='text-muted-foreground'>({nfts?.length})</p>
        ) : null}
      </div>
      {nfts?.length ? (
        <div className='flex flex-col gap-2 pl-4'>
          {nfts.map((nft) => (
            <div className='flex gap-2' key={nft.address}>
              <img
                src={nft.image}
                className='size-7 object-contain rounded-md overflow-hidden'
              />
              <article className='flex items-center'>
                <a href={`${nft.address}`}>
                  {nft.name}{' '}
                  <span className=' absolute right-4'>{`${
                    balances![nft.address] || 0
                  }`}</span>
                </a>
              </article>
            </div>
          ))}
        </div>
      ) : null}
    </div>
  );
};

export default TokenItem;
```
